HTML
<div class="media">
  <div class="media__header">Media Header</div>
  <div class="media__image-wrap media__image-wrap--left">
    <a href="">
      <img src="" alt="Square Thumbnail">
    </a>
  </div>
  <div class="media__content-wrap">
  <div class="media__content">
    <h3 class="media__title">Media Title</h3>
    <p class="media__text">Lorem ipsum dolor sit amet...</p>
    <!-- Nested `.media` -->
    </div>
  </div>

</div>
Mixins
// COMPONENT MIXINS: MEDIA OBJECT
// ==============================
// File: codebase/scss/07_component-mixins/_media-object.scss

@mixin media {
  @include clearfix;
}

@mixin media__header {
  display: block;
  h2,
  h3,
  h4 {
    margin: 0;
    padding: 0;
  }
}

@mixin media__footer {
  @include clearfix;
  display: block;
}

@mixin media__image-wrap--left {
  float: left;
  & > a,
  & > img {
    display: inline-block;
  }
}
@mixin media__image-wrap--right {
  float: right;
  & > a,
  & > img {
    display: inline-block;
  }
}
@mixin media__image-wrap--top {
  & > a,
  & > img {
    display: block;
    text-align: center;
  }
}

@mixin media__content-wrap {
  overflow: hidden;
}

@mixin media__title {
  margin-top: $deco-spacing-sm;
}
Classes
// COMPONENT CLASS-BUILDER: MEDIA OBJECT
// =====================================
// File: codebase/scss/11_component-classes/_media-object.scss

.media {
  @include media;
}

.media__header {
  @include media__header;
}

.media__image-wrap {  
  &--left {
    @include media__image-wrap--left;
  }
  &--right {
    @include media__image-wrap--right;
  }
  &--top {
    @include media__image-wrap--top;
  }
}

.media__content-wrap {
  @include media__content-wrap;
}

.media__title {
  @include media__title;
}

.media__footer {
  @include media__footer;
}

Media object images can have media__image-wrap--left, media__image-wrap--right or media__image-wrap--center. Use media__image-wrap--center if you require a "card" layout as below (these media objects below are wrapped in grid columns).
