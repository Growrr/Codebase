Initial icon

.initial-icon

For use in notifications, etc. Similar to dropcap, but using i:first-child instead of ::first-letter. The enlarged icon spans two lines of text.

Mixins
// UTILITY: TYPOGRAPHY MIXINS
// ==========================
// File: codebase/scss/06_utility-mixins/_typography.scss

// Lead paragraph

@mixin lead {
  font-size: $base-font-size * 1.2;
}

// Drop cap

@mixin dropcap {
  &::first-letter {
    display: inline-block;
    float: left;
    padding-right: 4px;
    font-size: 575%;
    font-weight: 300;
    line-height: 75%;
  }
}

@mixin initial-icon {
  i:first-child {
    display: inline-block;
    float: left;
    padding-right: 20px;
    font-size: 340%;
    font-weight: 300;
    line-height: 90%;
  }
  min-height: $base-line-height * 2;
}

// Text alignment

@mixin text--left {
  text-align: left;
}

@mixin text--center {
  text-align: center;
}

@mixin text--right {
  text-align: right;
}
