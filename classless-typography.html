Variables
// DEFAULT VARIABLES: TYPOGRAPHY
// =============================
// File: codebase/scss/01_default-variables/_typography.scss

$base-font-family:
  -apple-system, // Apple
  BlinkMacSystemFont, // Chrome for OSX (San Francisco)
  "Segoe UI", // Windows
  "Roboto", "Oxygen-Sans", "Ubuntu", "Cantarell", // Android, Chrome, Linux
  "Helvetica Neue", sans-serif // Fallback
  !default;
$base-font-size: 1rem !default;
$base-line-height: 1.5 !default;
$base-line-height-rem: $base-font-size * $base-line-height !default;

$base-text-color: $deco-charcoal !default;
$base-text-color-muted: $deco-dark-gray !default;
$base-text-color-hyperlink: $deco-blue !default;
$base-text-color-hyperlink-hover: $deco-dark-blue !default;
Base Styles
// DEFAULT STYLES: TYPOGRAPHY
// ==========================
// File: codebase/scss/04_base-styles/_typography.scss

body {
  font-family: $base-font-family;
  font-size: $base-font-size;
  line-height: $base-line-height;
  color: $base-text-color;
  text-rendering: optimizeLegibility;
}

h1, h2, h3, h4, h5, h6 {
  margin-top: 0;
  margin-bottom: $base-line-height-rem;
  padding: 0;
  line-height: 1.3;
}

p, ol, ul, dl {
  margin-top: 0;
  margin-bottom: $base-line-height-rem;
}

h1 { font-size: $base-font-size * 2.125; }
h2 { font-size: $base-font-size * 1.75; }
h3 { font-size: $base-font-size * 1.5; }
h4 { font-size: $base-font-size * 1.25; }
h5 { font-size: $base-font-size * 1.1; }
h6 { font-size: $base-font-size * 1; }

@include break__sm--up {
  h1 { font-size: $base-font-size * 3; }
  h2 { font-size: $base-font-size * 2.25; }
  h3 { font-size: $base-font-size * 1.75; }
  h4 { font-size: $base-font-size * 1.35; }
}

hr {
  height: 0;
  margin: $deco-spacing-lg 0;
  border: 0;
  border-top: $deco-border-thin-solid;
}

blockquote {
  margin: $base-line-height-rem 0;
  border-left: $deco-border-thick-solid;
  padding: $deco-spacing-md $deco-spacing-md $deco-spacing-md $deco-spacing-lg;
  font-size: $base-font-size * 1.2;
}

a {
  color: $base-text-color-hyperlink;
  text-decoration: none;
  &:hover {
    color: $base-text-color-hyperlink-hover;
    text-decoration: underline;
  }
  &:active,
  &:focus {
    outline: none;
  }
}
