Mixins
// COMPONENT MIXINS: MENUS
// =======================
// File: codebase/scss/07_component-mixins/_menus.scss

@mixin menu {
  list-style-type: none;
  margin: 0;
  padding: 0;
}

@mixin menu__item {
  list-style-type: none;
}

@mixin menu__link {
  display: inline-block;
  padding: $deco-spacing-sm 0;
}

@mixin menu__link--block {
  display: block;
  padding: $deco-spacing-block;
  &:hover {
    text-decoration: none;
    background: $deco-menu-block-hover-background;
  }
}

@mixin menu--horizontal {
  padding-left: 0;
  li {
    display: inline-block;
    padding-right: $deco-spacing-md;
    &last-of-type {
      padding-right: 0;
    }
  }
}
Classes
// COMPONENT CLASS-BUILDER: MENUS
// ==============================
// File: codebase/scss/11_component-classes/_menus.scss

// Basic links
.menu {
  @include menu;
  li {
    @include menu__item;
  }
  a {
    @include menu__link;
  }
  // Child menus
  .menu {
    margin-left: $deco-spacing-md;
  }
}

// Block links
.menu--block {
  a {
    @include menu__link--block;
  }
  // Child menus
  .menu--block {
    margin-left: 0;
    a {
      padding-left: $deco-spacing-lg;
    }
  }
}

// Horizontal menu (a.k.a. menubar)
.menu--horizontal,
.menubar {
  @include menu--horizontal;
  &.menu--block {
    li {
      padding-right: 0;
    }
  }
}
Menu is displayed as: <ul> classes: Alias <ul> classes:
Vertical inline-block .menu
Vertical block .menu.menu--block
Horizontal inline-block .menu.menu--horizontal .menubar
Horizontal block .menu.menu--horizontal.menu--block .menubar.menubar--block
